ตัวอย่างการอ่านไฟล์ txt

อันนี้พี่ยกตัวอย่างมา ถ้าครูสอนเขียนแบบนี้ก็ทำความเข้าใจตามโค้ดข้างล่างได้เลยนะ (ลองเอาไปรันดู)

#include <stdio.h>

int main()
{
    FILE *file;  // ตัวแปรสำหรับเก็บข้อมูลไฟล์
    char line[256];  // ตัวแปรเก็บแต่ละบรรทัดที่อ่าน

    // เปิดไฟล์ในโหมดอ่าน ("r")
    file = fopen("studentscore.txt", "r");

    // ตรวจสอบว่าเปิดไฟล์สำเร็จหรือไม่
    if (file == NULL) {
        printf("ไม่สามารถเปิดไฟล์ได้\n");
        return 1;  // ออกจากโปรแกรมถ้าเปิดไฟล์ไม่สำเร็จ
    }

    // อ่านไฟล์ทีละบรรทัด
    while (fgets(line, sizeof(line), file) != NULL) {
        printf("%s", line);  // แสดงบรรทัดที่อ่าน
    }

    // ปิดไฟล์
    fclose(file);

    return 0;
}

ขั้นตอนการประกอบโค้ดตามแบบฝึกหัด

ให้นักเรียนเขียนโปรแกรมอ่านไฟล์ studentscore.txt แล้วประกาศโครงสร้าง student เก็บข้อมูลนักเรียน ได้แก่ ชื่อ คะแนน math, reading, writing จากนั้น สร้างเมนูย่อย ได้แก่

1. เมนูสอบถามคะแนนแต่ละวิชา โดยป้อนชื่อนักเรียน  ถ้าหากพบชื่อ ให้ปริ้นคะแนนทั้ง 3 วิชา  ถ้าไม่พบ Not Found

2. เมนูแสดงค่าสถิติเบื้องต้น แต่ละวิชา (Min , Max, Average)

การส่งงาน ให้นักเรียน ถ่าย vdo อธิบายโค้ดแต่ละบรรทัด พร้อมทั้งแสดงผลลัพธ์

Step 1 : อ่านไฟล์ studentscore.txt แล้วประกาศโครงสร้าง student เก็บข้อมูลนักเรียน ได้แก่ ชื่อ คะแนน math, reading, writing 

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// กำหนดโครงสร้าง student
struct student {
    char name[50];
    int math;
    int reading;
    int writing;
};

int main() {
    FILE *file;
    struct student students[1000];  // อาเรย์เก็บข้อมูลนักเรียน
    int student_count = 0;  // ตัวแปรนับจำนวนของนักเรียนที่อ่านได้

    // เปิดไฟล์ studentscore.txt ในโหมดอ่าน
    file = fopen("studentscore.txt", "r");
    if (file == NULL) {
        printf("ไม่สามารถเปิดไฟล์ได้\n");
        return 1;  // ออกจากโปรแกรมหากไม่สามารถเปิดไฟล์ได้
    }

    // อ่านข้อมูลจากไฟล์ทีละบรรทัดและเก็บลงในโครงสร้าง
    while (fscanf(file, "%s %d %d %d", students[student_count].name, 
                   &students[student_count].math, 
                   &students[student_count].reading, 
                   &students[student_count].writing) == 4) {
        student_count++;
    }

    // ปิดไฟล์หลังการอ่าน
    fclose(file);

    // แสดงผลข้อมูลนักเรียนที่เก็บในอาเรย์
    printf("ข้อมูลนักเรียน:\n");
    for (int i = 0; i < student_count; i++) {
        printf("ชื่อ: %s\n", students[i].name);
        printf("คะแนนคณิตศาสตร์: %d\n", students[i].math);
        printf("คะแนนการอ่าน: %d\n", students[i].reading);
        printf("คะแนนการเขียน: %d\n", students[i].writing);
        printf("\n");
    }

    return 0;
}

Step 3 Final : สร้างเมนูย่อย→ เมนูแสดงค่าสถิติเบื้องต้น แต่ละวิชา (Min , Max, Average)

ค่า Min หาได้จาก คะแนนที่น้อยที่สุดของวิชานั้นๆ

ค่า Max หาได้จาก คะแนนที่มากที่สุดของวิชานั้นๆ

ค่า Average คือ ค่าเฉลี่ย หาได้จากผลรวมของคะแนนแต่ละวิชาหารด้วยจำนวนนักเรียนทั้งหมด

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

// กำหนดโครงสร้าง student
struct student {
    char name[50];
    int math;
    int reading;
    int writing;
};

int main() {
    FILE *file;
    struct student students[1000];  // อาเรย์เก็บข้อมูลนักเรียน
    int student_count = 0;  // ตัวแปรนับจำนวนของนักเรียนที่อ่านได้

    // เปิดไฟล์ studentscore.txt ในโหมดอ่าน
    file = fopen("studentscore.txt", "r");
    if (file == NULL) {
        printf("ไม่สามารถเปิดไฟล์ได้\n");
        return 1;  // ออกจากโปรแกรมหากไม่สามารถเปิดไฟล์ได้
    }

    // อ่านข้อมูลจากไฟล์ทีละบรรทัดและเก็บลงในโครงสร้าง
    while (fscanf(file, "%s %d %d %d", students[student_count].name, 
                   &students[student_count].math, 
                   &students[student_count].reading, 
                   &students[student_count].writing) == 4) {
        student_count++;
    }

    // ปิดไฟล์หลังการอ่าน
    fclose(file);
    
    // แสดงเมนู
    int choice;
    do {
        printf("1. สอบถามคะแนนนักเรียน\n");
        printf("2. แสดงค่าสถิติเบื้องต้น\n");
        printf("3. ออกจากโปรแกรม\n");
        printf("กรุณาเลือกเมนู: ");
        scanf("%d", &choice);
    
        switch (choice) {
            // step 2 : เมนูสอบถามคะแนนแต่ละวิชา โดยป้อนชื่อนักเรียน ถ้าหากพบชื่อ ให้ปริ้นคะแนนทั้ง 3 วิชา ถ้าไม่พบ Not Found
            case 1: {
                // รับค่าชื่อนักเรียน
                char name[50];
                printf("\nกรุณากรอกชื่อของนักเรียน: ");
                scanf("%s", name);
                
                // กำหนดตัวแปรสำหรับเก็บข้อมูลนักเรียน ให้เริ่มต้นที่ -1
                // ถ้าพบข้อมูลนักเรียนคนนั้นจะจัดเก็บลำดับของนักเรียนใส่ในค่านี้ 
                // แต่ถ้าไม่พบข้อมูลนักเรียนที่กรอกมาจะคงค่าเดิมคือ -1
                int i = -1;
                // วนลูปเพื่อหาข้อมูลนักเรียน
                for (int j = 0; j < student_count; j++) {
                    if (strcmp(students[j].name, name) == 0) {
                       i = j;
                       break; // เมื่อเจอแล้วให้หยุดหา
                    }
                }
                
                // หลังจากหาข้อมูลเรียบร้อยแล้ว มาเช็คต่อว่าค่า i คืออะไรถ้าไม่ใช่ -1 แสดงว่าหาข้อมูลนักเรียนเจอ
                if (i != -1) {
                    printf("ชื่อ: %s\n", students[i].name);
                    printf("คะแนนคณิตศาสตร์: %d\n", students[i].math);
                    printf("คะแนนการอ่าน: %d\n", students[i].reading);
                    printf("คะแนนการเขียน: %d\n", students[i].writing);
                    printf("\n");
                } else {
                    // ถ้าไม่พบชื่อนักเรียน
                    printf("Not Found\n");
                }
                
                break;
            }
            
            // step 3: สร้างเมนูย่อย→ เมนูแสดงค่าสถิติเบื้องต้น แต่ละวิชา (Min , Max, Average)
            case 2: {
                
                // กำหนดให้ค่าเริ่มต้นของ min กับ max เป็นนักเรียนคนแรก
                int min_math = students[0].math, max_math = students[0].math;
                int min_reading = students[0].reading, max_reading = students[0].reading;
                int min_writing = students[0].writing, max_writing = students[0].writing;
                
                // กำหนดค่า sum คือผลรวมของคะแนนแต่ละวิชา
                int sum_math = 0, sum_reading = 0, sum_writing = 0;
                
                for (int i = 0; i < student_count; i++) {
                    
                    // คำนวณค่าสูงสุดและต่ำสุดของคะแนนแต่ละวิชา
                    if (students[i].math < min_math) min_math = students[i].math;
                    if (students[i].math > max_math) max_math = students[i].math;
                    if (students[i].reading < min_reading) min_reading = students[i].reading;
                    if (students[i].reading > max_reading) max_reading = students[i].reading;
                    if (students[i].writing < min_writing) min_writing = students[i].writing;
                    if (students[i].writing > max_writing) max_writing = students[i].writing;

                    // คำนวณผลรวมของคะแนนแต่ละวิชา
                    sum_math += students[i].math;
                    sum_reading += students[i].reading;
                    sum_writing += students[i].writing;
                }
                // คำนวณค่าเฉลี่ย
                float avg_math = (float)sum_math / student_count;
                float avg_reading = (float)sum_reading / student_count;
                float avg_writing = (float)sum_writing / student_count;

                // แสดงผลค่าสถิติ
                printf("\nค่าสถิติเบื้องต้น:\n");
                printf("วิชาคณิตศาสตร์: Min = %d, Max = %d, Average = %.2f\n", min_math, max_math, avg_math);
                printf("วิชาการอ่าน: Min = %d, Max = %d, Average = %.2f\n", min_reading, max_reading, avg_reading);
                printf("วิชาการเขียน: Min = %d, Max = %d, Average = %.2f\n", min_writing, max_writing, avg_writing);
                
                break;
            }
        }
    
    } while (choice != 3);
        
}
